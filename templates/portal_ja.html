      PREFIX oa: <http://www.w3.org/ns/oa#>
       
      SELECT  distinct ?hpo FROM <https://pubcasefinder.dbcls.jp/> WHERE{
        ?an rdf:type oa:Annotation ;
            oa:hasTarget <http://identifiers.org/mim/100050> ;
            oa:hasBody ?hpo.
      }

      SELECT  distinct ?gene FROM <https://pubcasefinder.dbcls.jp/> WHERE{
        ?as sio:SIO_000628 <http://identifiers.org/mim/616645> ;
            sio:SIO_000628 ?gene .

      }
